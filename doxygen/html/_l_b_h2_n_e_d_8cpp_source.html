99 std::vector<std::vector<double>> BLH2XYZ::mul_mat(std::vector<std::vector<double>> mat_a, std::vector<std::vector<double>> mat_b)
100 {
101  unsigned int i;
102  unsigned int j;
103  unsigned int k;
104  std::vector<std::vector<double>> mat(3, std::vector<double>(3, 0.0));
105 
106  try {
107  for (i = 0; i < 3; ++i) {
108  for (j = 0; j < 3; ++j) {
109  for (k = 0; k < 3; ++k) {
110  mat[i][j] += mat_a[i][k] * mat_b[k][j];
111  }
112  }
113  }
114  }
115  catch (...) {
116  throw;
117  }
118 
119  return mat;

152 BLH2XYZ::CoordX BLH2XYZ::blh2ecef(CoordB c_src)
153 {
154  CoordX ecef;
155 
156  try {
157  ecef.x = (n(c_src.b) + c_src.h) * cos(c_src.b * kPi180) * cos(c_src.l * kPi180);
158  ecef.y = (n(c_src.b) + c_src.h) * cos(c_src.b * kPi180) * sin(c_src.l * kPi180);
159  ecef.z = (n(c_src.b) * (1.0 - kE2) + c_src.h) * sin(c_src.b * kPi180);
160  }
161  catch (...) {
162  throw;
163  }
164 
165  return ecef;
166 }

168 BLH2XYZ::CoordE BLH2XYZ::blh2enu(CoordB c_0, CoordB c_1)
169 {
170  CoordX ecef_x_0;
171  CoordX ecef_x_1;
172  CoordX ecef_x;
173  std::vector<std::vector<double>> mat_0(3, std::vector<double>(3, 0.0));
174  std::vector<std::vector<double>> mat_1(3, std::vector<double>(3, 0.0));
175  std::vector<std::vector<double>> mat_2(3, std::vector<double>(3, 0.0));
176  std::vector<std::vector<double>> mat;
177  CoordE enu;
178 
179  try {
180  // BLH -> ECEF
181  ecef_x_0 = blh2ecef(c_0);
182  ecef_x_1 = blh2ecef(c_1);
183  ecef_x = { ecef_x_1.x - ecef_x_0.x,
184  ecef_x_1.y - ecef_x_0.y,
185  ecef_x_1.z - ecef_x_0.z };
186  // 回転行列生成
187  mat_0 = mat_z(90.0);
188  mat_1 = mat_y(90.0 - c_0.b);
189  mat_2 = mat_z(c_0.l);
190  mat = mul_mat(mul_mat(mat_0, mat_1), mat_2);
191  enu = rotate(mat, ecef_x);
192  }
193  catch (...) {
194  throw;
195  }
196 
197  return enu;
198 }

200 FlightData BLH2XYZ::BLH2NED(XPlaneData XPlaneData_, float L0, float B0, float H0)
201 {
202  BLH2XYZ BLH2XYZ_; //实例化一个坐标转换工具人
203  FlightData FlightData_;
204  CoordB blh_0; // 変換前の座標（基準）
205  CoordB blh_1; // 変換前の座標（対象）
206  CoordE enu; // 変換後の座標
207  blh_0.b = B0;
208  blh_0.l = L0;
209  blh_0.h = H0;
210  blh_1.b = XPlaneData_.BLH[0];
211  blh_1.l = XPlaneData_.BLH[1];
212  blh_1.h = XPlaneData_.BLH[2] * 0.3048;
213 
214  enu = BLH2XYZ_.blh2enu(blh_0, blh_1); // BLH -> ENU
215 
216  FlightData_.position[0] = (float)enu.n;
217  FlightData_.position[1] = (float)enu.e;
218  FlightData_.position[2] = (float)-enu.u;
219  FlightData_.attitude[0] = (float)XPlaneData_.attitude[0] / 57.3f; //pitch(rad)
220  FlightData_.attitude[1] = (float)XPlaneData_.attitude[1] / 57.3f; //roll(rad)
221  FlightData_.attitude[2] = (float)XPlaneData_.attitude[2] / 57.3f; //heading(rad)
222  return FlightData_;
223 }
