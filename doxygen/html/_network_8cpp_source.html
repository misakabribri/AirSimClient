127 FlightData Network::DecodeByteMATLAB(std::vector<char>& recv_buff)
128 {
129  FlightData flight_data;
130  // std::string str = recv_buff.data();
131  if (recv_buff[0] == 'D' && recv_buff[1] == 'A') // 数据帧头"DATA@"校验
132  {
133  double* Data[6]; // 共计6个float需要读取
134  int ByteCounter = 5; // 除去帧头，从第6个byte开始
135  for (int N = 0; N <= 5; N++) {
136  Data[N] = (double*)&recv_buff[ByteCounter];
137  ByteCounter = ByteCounter + 8;
138  }
139  flight_data.position[0] = (float)*Data[0]; // X
140  flight_data.position[1] = (float)*Data[1]; // Y
141  flight_data.position[2] = (float)-*Data[2]; // Z
142  flight_data.attitude[0] = (float)*Data[3]; // theta
143  flight_data.attitude[1] = (float)*Data[4]; // phi
144  flight_data.attitude[2] = (float)*Data[5]; // psi
145  }
146  return flight_data;

149 XPlaneData Network::DecodeByteXPlane(std::vector<char>& recv_buff)
150 {
151  XPlaneData XPlaneData_;
152  uint32_t* IndexID[5];
153  float* XPlaneData[5][8]; // 分配空间，X-Plane中勾选了5行数据，每行1个unin32+8个float
154  if (recv_buff[0] == 'D' && recv_buff[1] == 'A') // 数据帧头"DATA@"校验
155  {
156  int ByteCounter = 5; // 数据帧头长度
157  for (int i = 0; i < 5; i++) // 共计5个数据包
158  {
159  for (int j = 0; j < 9; j++) // 每个数据包9个数据（1+8）
160  {
161  if (j < 1) {
162  IndexID[i] = (uint32_t*)&recv_buff[ByteCounter]; // 数据包ID，含义对应X-Plane的Data界面
163  }
164  else {
165  XPlaneData[i][j - 1] = (float*)&recv_buff[ByteCounter]; // 记录数据包数据，这里注意是j-1，因为XPlaneData数组仍然从0开始计数
166  }
167  ByteCounter = ByteCounter + 4;
168  }
169  }
170  }
171  XPlaneData_.Vtrue = *XPlaneData[0][7]; // Vtrue(mph)
172  XPlaneData_.angular_velocities[0] = *XPlaneData[1][0]; // Q(rad/s)
173  XPlaneData_.angular_velocities[1] = *XPlaneData[1][1]; // P(rad/s)
174  XPlaneData_.angular_velocities[2] = *XPlaneData[1][2]; // R(rad/s)
175  XPlaneData_.attitude[0] = *XPlaneData[2][0]; // pitch(deg)
176  XPlaneData_.attitude[1] = *XPlaneData[2][1]; // roll(deg)
177  XPlaneData_.attitude[2] = *XPlaneData[2][2]; // heading(deg)
178  XPlaneData_.wind_angle[0] = *XPlaneData[3][0]; // alpha(deg)
179  XPlaneData_.wind_angle[1] = *XPlaneData[3][1]; // beta(deg)
180  XPlaneData_.BLH[0] = *XPlaneData[4][0]; // lat(deg)
181  XPlaneData_.BLH[1] = *XPlaneData[4][1]; // lon(deg)
182  XPlaneData_.BLH[2] = *XPlaneData[4][2]; // alt(ft)
183  return XPlaneData_;
184 }
