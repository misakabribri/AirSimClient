121 std::vector<uchar> UE4Helper::ImageEncoder(ImageCaptureResult& image_capture_result)
122 {
123  int Width = image_capture_result.response.at(0).width;
124  int Height = image_capture_result.response.at(0).height;
125  std::vector<uchar> encoded_image(Width * Height * 3);
126  cv::Mat cv_image(Height, Width, CV_8UC(4));
127  int idx = 0;
128  for (int x = 0; x < Height; x++) {
129  for (int y = 0; y < Width; y++) {
130  cv_image.at<cv::Vec4b>(x, y)[0] = image_capture_result.response.at(0).image_data_uint8[idx]; //B
131  cv_image.at<cv::Vec4b>(x, y)[1] = image_capture_result.response.at(0).image_data_uint8[idx + 1]; //G
132  cv_image.at<cv::Vec4b>(x, y)[2] = image_capture_result.response.at(0).image_data_uint8[idx + 2]; //R
133  cv_image.at<cv::Vec4b>(x, y)[3] = 0; //alpha
134  idx += 3;
135  }
136  }
137  vector<int> encode_param = vector<int>(2);
138  encode_param[0] = CV_IMWRITE_JPEG_QUALITY;
139  encode_param[1] = 70;
140  if (cv_image.data) //防止图片内容为空，导致opencv错误
141  {
142  //cv::imshow("pic-origin", cv_image);
143  //cv::waitKey(1);
144  cv::imencode(".jpg", cv_image, encoded_image, encode_param);
145  }
146  return encoded_image;
147 }
